create function w(@birthday datetime)
returns varchar(10)
as
begin
	if year(@birthday) between 1980 and 1989
		return '80s'
	if year(@birthday) between 1990 and 1999
		return '90s'
	else
		return 'too old'
	return ''
end

select dbo.w('1999-01-01')
