from student

	select classno
	from student

	group by classno,sex
	having COUNT(sno) >=3;
	
select classno
from student
where sex = '男'
group by classno
having COUNT(sno) <3
union
	select distinct classno
	from student
	where classno not in (
		select distinct classno
		from student 
		where sex = '男'
	);
